    03 —

    Réalisation & Soudure

    01
    Préparation des points

    Trouver un endroit sécurisé pour utiliser le fer à souder.

    ⚠️Si vous n'avez jamais soudé, consultez un tutoriel avant de commencer — la soudure mal maîtrisée peut être dangereuse.

    02
    La Soudure

    Soudure le chargeur femmelle au petit Module ainsi:

  • Fil Rouge de la batterie (+) ➔ Broche B+ (ou BAT+) du petit module Type-C.
  • Fil Noir de la batterie (-) ➔ Broche B- (ou BAT-) du module Type-C.

    • Puis la carte ESP32 CAM au Module ainsi:
  • Broche OUT+ (ou VO+) du module ➔ Broche 5V de l'ESP32-CAM. (Surtout pas sur la broche 3.3V !)
  • Broche OUT- (ou VO-) du module ➔ Broche GND (terre) de l'ESP32-CAM.

    • 03
    Test Finale

    Insertion la batterie Li-Po

    ⚠️ Attention — une mauvaise soudure peut provoquer un court-circuit, un incendie, voire l'explosion de la batterie. Vérifiez deux fois avant de brancher.

    Code Source

    Le firmware tourne sur l'ESP32-CAM AI Thinker, écrit en C++ via le framework Arduino. Il gère la capture photo au réveil, l'application de 20 filtres différents et l'enregistrement JPEG sur carte SD — le tout en deep sleep pour économiser la batterie.

    Comment installer

    5 étapes pour téléverser le code sur votre ESP32-CAM.

    01
    Installer Arduino IDE

    Télécharger et installer Arduino IDE 2.x depuis le site officiel d'Arduino. C'est l'éditeur dans lequel on colle et téléverse le code.

    02
    Ajouter le support ESP32

    Aller dans Fichier → Préférences, puis coller l'URL suivante dans le champ "URL de gestionnaire de cartes supplémentaires" :

    https://raw.githubusercontent.com/espressif/arduino-esp32/gh-pages/package_esp32_index.json
    03
    Installer le paquet Espressif

    Dans le Gestionnaire de cartes (icône à gauche), rechercher "esp32" et installer "esp32 by Espressif Systems". L'opération peut prendre quelques minutes.

    04
    Sélectionner la bonne carte

    Aller dans Outils → Carte → esp32 et choisir "AI Thinker ESP32-CAM".

    05
    Aucune bibliothèque à ajouter ✓

    Bonne nouvelle : toutes les bibliothèques utilisées (esp_camera.h, SD_MMC.h, EEPROM.h…) sont déjà incluses dans le paquet Espressif installé à l'étape 3. Rien d'autre à télécharger.

    Code complet

    main.ino 
    #include "esp_camera.h"
#include "FS.h"
#include "SD_MMC.h"
#include "EEPROM.h"

// Pin definition pour ESP32-CAM
#define PWDN_GPIO_NUM     32
#define RESET_GPIO_NUM    -1
#define XCLK_GPIO_NUM      0
#define SIOD_GPIO_NUM     26
#define SIOC_GPIO_NUM     27
#define Y9_GPIO_NUM       35
#define Y8_GPIO_NUM       34
#define Y7_GPIO_NUM       39
#define Y6_GPIO_NUM       36
#define Y5_GPIO_NUM       21
#define Y4_GPIO_NUM       19
#define Y3_GPIO_NUM       18
#define Y2_GPIO_NUM        5
#define VSYNC_GPIO_NUM    25
#define HREF_GPIO_NUM     23
#define PCLK_GPIO_NUM     22

int pictureNumber = 1;
int currentFilter = 0;

void appliquerFiltre(sensor_t *s, int f) {
  s->set_special_effect(s, 0); s->set_brightness(s, 0);
  s->set_contrast(s, 0); s->set_saturation(s, 0); s->set_wb_mode(s, 0);
  switch(f) {
    case 0:  break;
    case 1:  s->set_special_effect(s, 2); break; // N&B
    case 2:  s->set_special_effect(s, 6); break; // Sépia
    case 3:  s->set_special_effect(s, 1); break; // Négatif
    case 4:  s->set_special_effect(s, 3); break; // Rouge
    case 5:  s->set_special_effect(s, 4); break; // Vert
    case 6:  s->set_special_effect(s, 5); break; // Bleu
    case 7:  s->set_contrast(s, 2); break;        // Contraste+
    case 8:  s->set_contrast(s, -2); s->set_saturation(s, -2); break; // Délavé
    case 9:  s->set_brightness(s, 2); break;      // High Key
    case 10: s->set_brightness(s, -2); s->set_contrast(s, 2); break;  // Low Key
    case 11: s->set_special_effect(s, 2); s->set_contrast(s, 2); break;  // N&B Hard
    case 12: s->set_special_effect(s, 6); s->set_contrast(s, 1); break;  // Sépia Hard
    case 13: s->set_wb_mode(s, 3); break; // Froid
    case 14: s->set_wb_mode(s, 4); break; // Chaud
    case 15: s->set_special_effect(s, 1); s->set_contrast(s, 2); break;  // Négatif Hard
    case 16: s->set_special_effect(s, 3); s->set_brightness(s, -2); break; // Rouge Dark
    case 17: s->set_special_effect(s, 5); s->set_brightness(s, 1); break;  // Bleu Light
    case 18: s->set_special_effect(s, 4); s->set_contrast(s, 2); break;    // Vert Matrix
    case 19: s->set_special_effect(s, 2); s->set_contrast(s, -2); break;   // N&B Soft
  }
}

void setup() {
  Serial.begin(115200);
  pinMode(4, OUTPUT);
  digitalWrite(4, LOW);

  EEPROM.begin(512);
  EEPROM.get(0, pictureNumber);
  EEPROM.get(4, currentFilter);

  if (pictureNumber <= 0 || pictureNumber > 10000) pictureNumber = 1;
  if (currentFilter < 0 || currentFilter > 19)    currentFilter = 0;

  if (esp_sleep_get_wakeup_cause() == ESP_SLEEP_WAKEUP_EXT0) {

    // === BOUTON IO0 : appui court / long ===
    pinMode(0, INPUT);
    unsigned long pressTime = millis();
    bool isLongPress = false;

    while (digitalRead(0) == LOW) {
      if (millis() - pressTime > 800) { isLongPress = true; break; }
      delay(10);
    }

    if (isLongPress) {
      // Appui long → reset du filtre
      currentFilter = 0;
      digitalWrite(4, HIGH); delay(800); digitalWrite(4, LOW);
      while (digitalRead(0) == LOW) { delay(10); }
    } else {
      // Appui court → filtre suivant
      currentFilter++;
      if (currentFilter > 19) currentFilter = 0;
      for (int i = 0; i <= currentFilter; i++) {
        digitalWrite(4, HIGH); delay(2);
        digitalWrite(4, LOW);  delay(250);
      }
    }

    EEPROM.put(4, currentFilter);
    EEPROM.commit();

  } else {

    // === MODE RST : prise de photo ===
    camera_config_t config;
    config.ledc_channel = LEDC_CHANNEL_0; config.ledc_timer = LEDC_TIMER_0;
    config.pin_d0 = Y2_GPIO_NUM; config.pin_d1 = Y3_GPIO_NUM;
    config.pin_d2 = Y4_GPIO_NUM; config.pin_d3 = Y5_GPIO_NUM;
    config.pin_d4 = Y6_GPIO_NUM; config.pin_d5 = Y7_GPIO_NUM;
    config.pin_d6 = Y8_GPIO_NUM; config.pin_d7 = Y9_GPIO_NUM;
    config.pin_xclk  = XCLK_GPIO_NUM;  config.pin_pclk  = PCLK_GPIO_NUM;
    config.pin_vsync = VSYNC_GPIO_NUM; config.pin_href  = HREF_GPIO_NUM;
    config.pin_sscb_sda = SIOD_GPIO_NUM; config.pin_sscb_scl = SIOC_GPIO_NUM;
    config.pin_pwdn  = PWDN_GPIO_NUM;  config.pin_reset = RESET_GPIO_NUM;
    config.xclk_freq_hz = 20000000;
    config.pixel_format = PIXFORMAT_JPEG;
    config.frame_size   = FRAMESIZE_UXGA;
    config.jpeg_quality = 10;
    config.fb_count     = 1;

    if (esp_camera_init(&config) == ESP_OK) {
      sensor_t *s = esp_camera_sensor_get();
      if (s != NULL) appliquerFiltre(s, currentFilter);

      if (SD_MMC.begin("/sdcard", true)) {
        digitalWrite(4, HIGH);
        delay(1500);
        camera_fb_t *fb = esp_camera_fb_get();
        digitalWrite(4, LOW);

        if (fb) {
          String path;
          while (true) {
            path = "/photo_" + String(pictureNumber) + ".jpg";
            if (!SD_MMC.exists(path)) break;
            pictureNumber++;
          }
          File file = SD_MMC.open(path.c_str(), FILE_WRITE);
          if (file) {
            file.write(fb->buf, fb->len);
            EEPROM.put(0, pictureNumber + 1);
            EEPROM.commit();
          }
          file.close();
          esp_camera_fb_return(fb);
        }
      }
    }
  }

  esp_sleep_enable_ext0_wakeup(GPIO_NUM_0, 0);
  delay(500);
  esp_deep_sleep_start();
}

void loop() {}
